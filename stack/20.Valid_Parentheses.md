class Solution {
public:
    bool isValid(string s) {
        stack<char> stack;
        size_t length = s.length();
        if(length % 2 != 0)
            return false;
        size_t i = 0;
        stack.push(DUMMY_CHARACTER);
        while(i < length){
            if(stack.top() == '(' && s[i] == ')')
                stack.pop();
            else if(stack.top() == '{' && s[i] == '}')
                stack.pop();
            else if(stack.top() == '[' && s[i] == ']')
                stack.pop();
            else
                stack.push(s[i]);
            i++;
        }
        if(stack.size() != 1)
            return false;
        return true;
    }
private:
    const char DUMMY_CHARACTER = '_';
};
