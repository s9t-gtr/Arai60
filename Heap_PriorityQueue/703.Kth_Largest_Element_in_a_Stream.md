'''
class KthLargest {
public:
    KthLargest(int k, vector<int>& nums): k(k){
        for(int x: nums){
            pq.push(x);
            if(pq.size() > k)
                pq.pop();
        }
    }

    int add(int val) {
        pq.push(val);
        if(pq.size() > k)
            pq.pop();
        return pq.top();
    }
private:
    priority_queue<int, vector<int>, greater<int> > pq;
    int k;
};

'''
